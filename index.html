def broken_search(nums: list, target: int) -> int:
  """Search in a broken_array"""
  start = 0
  end = len(nums) - 1
  index = -1
  while start  <= end:
      mid = (start + end) // 2
      mid_value = nums[mid]
      if mid_value == target:
          index = mid
          break
      if nums[start]  <= mid_value:
          if nums[start]  <= target  <= mid_value:
              end = mid - 1
          else:
              start = mid + 1
      else:
          if mid_value  <= target  <= nums[end]:
              start = mid + 1
          else:
              end = mid - 1
  return index
